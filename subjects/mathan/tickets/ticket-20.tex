\question{Монотонность сумм Дарбу по разбиениям. Критерий интегрируемости}
\begin{note}
Разбиением будем называть конечное множество $P = \{x_0, x_1, \ldots, x_n\}$ такое, что $a = x_0 < x_1 < \ldots < x_n = b$. И записывать его как $\mathcal{T} = \{x_0, x_1, \ldots, x_n\}$
А его легкость считать как масикмальное расстояние между соседними точками: $\lambda(\mathcal{T}) = \max_{1 \leq k \leq n} (x_k - x_{k-1})$.
\end{note}

\begin{defn}[Суммы Дарбу]
  Пусть $f : [a, b] \to \R$ --- ограниченная функция, $P = \{x_0, x_1, \ldots, x_n\}$ --- разбиение отрезка $[a, b]$. Тогда \emph{нижняя сумма Дарбу} функции $f$ по разбиению $P$ --- это
  \[
    s(f, P) := \sum_{k=1}^n (x_k - x_{k-1})\cdot\inf_{x \in [x_{k-1}, x_k]} f(x).
  \]
  Аналогично, \emph{верхняя сумма Дарбу} функции $f$ по разбиению $P$ --- это
  \[
    S(f, P) := \sum_{k=1}^n (x_k - x_{k-1})\cdot\sup_{x \in [x_{k-1}, x_k]} f(x).
  \]
\end{defn}

\begin{theorem}[Монотонность сумм Дарбу по разбиениям] 
  Пусть $f : [a, b] \to \R$ --- ограниченная функция, $P$ и $Q$ --- разбиения отрезка $[a, b]$. Тогда
  \begin{enumerate}
    \item Если $P \subset Q$, то $s(f, P) \leq s(f, Q)$ и $S(f, P) \geq S(f, Q)$.
    \item Для всякого разбиения $P$ выполняется неравенство $s(f, P) \leq S(f, P)$.
  \end{enumerate}
\end{theorem}
\begin{proof}
  \textbf{Шаг 1.} Пусть $P \subset Q$. Тогда $Q$ получается из $P$ добавлением новых точек. Достаточно показать, что при добавлении одной новой точки $s(f, P)$ не убывает, а $S(f, P)$ не возрастает.

  Пусть $P = \{x_0, x_1, \ldots, x_n\}$ и $Q = P \cup \{y\}$, где $y \in (x_{k-1}, x_k)$ для некоторого $k$. Тогда
  \[
    s(f, Q) = s(f, P) - (x_k - x_{k-1})\cdot\inf_{x \in [x_{k-1}, x_k]} f(x) + (y - x_{k-1})\cdot\inf_{x \in [x_{k-1}, y]} f(x) + (x_k - y)\cdot\inf_{x \in [y, x_k]} f(x).
  \]
  Поскольку $\inf_{x \in [x_{k-1}, x_k]} f(x) \leq \inf_{x \in [x_{k-1}, y]} f(x)$ и $\inf_{x \in [x_{k-1}, x_k]} f(x) \leq \inf_{x \in [y, x_k]} f(x)$, то
  \[
    s(f, Q) \geq s(f, P).
  \]
  Аналогично для верхней суммы Дарбу:
  \[
    S(f, Q) = S(f, P) - (x_k - x_{k-1})\cdot\sup_{x \in [x_{k-1}, x_k]} f(x) + (y - x_{k-1})\cdot\sup_{x \in [x_{k-1}, y]} f(x) + (x_k - y)\cdot\sup_{x \in [y, x_k]} f(x).
  \]
  Поскольку $\sup_{x \in [x_{k-1}, x_k]} f(x) \geq \sup_{x \in [x_{k-1}, y]} f(x)$ и $\sup_{x \in [x_{k-1}, x_k]} f(x) \geq \sup_{x \in [y, x_k]} f(x)$, то
  \[
    S(f, Q) \leq S(f, P).
  \]
\end{proof}

\begin{theorem}[Критерий интегрируемости]
  Пусть $f : [a, b] \to \R$ --- ограниченная функция. Тогда $f$ интегрируема по Риману тогда и только тогда, когда
  \[
    \sup_{P} s(f, P) = \inf_{P} S(f, P).
  \]
\end{theorem}
