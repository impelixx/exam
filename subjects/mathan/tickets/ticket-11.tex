\question{Теорема о неявном отображении как следствие теоремы
о локальном диффеоморфизме}

% ═══════════════════════════════════════════════════════════════════════════
%  ОБОЗНАЧЕНИЯ
% ═══════════════════════════════════════════════════════════════════════════

\begin{note}[Обозначения]
  Пусть $F : \Omega \subset \mathbb{R}^{m+n} \to \mathbb{R}^n$, $\Omega$ открыто.
  Точки $\mathbb{R}^{m+n}$ пишем как $(x, y)$, где $x \in \mathbb{R}^m$,
  $y \in \mathbb{R}^n$. Матрицу Якоби разбиваем на два блока:
  \[
    F'_x \;:=\;
    \begin{pmatrix}
      \dfrac{\partial F_1}{\partial x_1} & \cdots &
      \dfrac{\partial F_1}{\partial x_m} \\[6pt]
      \vdots & \ddots & \vdots \\[2pt]
      \dfrac{\partial F_n}{\partial x_1} & \cdots &
      \dfrac{\partial F_n}{\partial x_m}
    \end{pmatrix},
    \qquad
    F'_y \;:=\;
    \begin{pmatrix}
      \dfrac{\partial F_1}{\partial y_1} & \cdots &
      \dfrac{\partial F_1}{\partial y_n} \\[6pt]
      \vdots & \ddots & \vdots \\[2pt]
      \dfrac{\partial F_n}{\partial y_1} & \cdots &
      \dfrac{\partial F_n}{\partial y_n}
    \end{pmatrix}.
  \]
  Тогда $\mathcal{D}F = \bigl(F'_x \;\big|\; F'_y\bigr)$.
  Блок $F'_y$ квадратный --- размера $n \times n$.
\end{note}

% ═══════════════════════════════════════════════════════════════════════════
%  ОПРЕДЕЛЕНИЕ НЕЯВНОГО ОТОБРАЖЕНИЯ
% ═══════════════════════════════════════════════════════════════════════════

\begin{defn}[Неявное отображение]
  Пусть $E \subset \mathbb{R}^{n+m}$, $X \subset \mathbb{R}^m$,
  $F : E \to \mathbb{R}^n$.
  Отображение $f : X \to \mathbb{R}^n$ называется \emph{неявным отображением},
  заданным уравнением $F(x, y) = 0$, если:
  \begin{enumerate}
    \item $\forall x \in X \;\Rightarrow\; (x,\, f(x)) \in E$;
    \item $F(x,\, f(x)) = 0 \quad \forall x \in X$.
  \end{enumerate}
  Геометрически: график $\mathrm{gr}\,f = \{(x, f(x)) : x \in X\}$
  лежит в множестве нулей $F$.
\end{defn}

\begin{note}[Мотивация]
  Глобально выразить $y$ как функцию от $x$ из уравнения $F(x,y) = 0$
  обычно нельзя: при одном $x$ может быть несколько значений $y$.
  Например, $x^2 + y^2 = 1$ при $x \in (-1,1)$ даёт два значения $y$.

  Но \emph{локально} --- в малой окрестности фиксированной точки ---
  один гладкий «лист» выделить можно. Условие, при котором это возможно:
  $\det F'_y \neq 0$. Если $\det F'_y = 0$, система уравнений по $y$
  вырождается и единственности нет.
\end{note}

%  ТЕОРЕМА О НЕЯВНОМ ОТОБРАЖЕНИИ

\begin{theorem}[О неявном отображении]
  \label{theorem:implicit}
  Пусть $\Omega \subset \mathbb{R}^{m+n}$ открыто, $F \in C^1(\Omega, \mathbb{R}^n)$.
  Пусть для некоторой точки $(x^0, y^0) \in \Omega$:
  \begin{enumerate}
    \item $F(x^0, y^0) = 0$;
    \item $\det F'_y(x^0, y^0) \neq 0$.
  \end{enumerate}
  Тогда существуют $\delta > 0$ и $\gamma > 0$ такие, что:
  \[
    \forall\, x^* \in Q_\delta(x^0)\quad
    \exists!\, y^* \in Q_\gamma(y^0) \;:\;
    F(x^*, y^*) = 0,
  \]
  где $Q_\delta(x^0) = \prod_{i=1}^m [x^0_i - \delta,\, x^0_i + \delta]$ ---
  куб со стороной $2\delta$.

  \medskip
  Тем самым определено отображение $f : Q_\delta(x^0) \to Q_\gamma(y^0)$.
  При этом $f \in C^1$ и выполнена формула:
  \[
    f'_x(x) \;=\;
    -\bigl(F'_y(x,\, f(x))\bigr)^{-1} \cdot F'_x(x,\, f(x)).
  \]
\end{theorem}

% ═══════════════════════════════════════════════════════════════════════════
%  ДОКАЗАТЕЛЬСТВО
% ═══════════════════════════════════════════════════════════════════════════

\begin{proof}
  \textbf{Ключевая идея.}
  Вводим вспомогательное отображение $\Phi$, которое переводит
  все нули $F$ в одну плоскость. Тогда $\Phi$ --- диффеоморфизм,
  а $\Phi^{-1}$ восстанавливает искомую функцию $f$.

  \bigskip
  \textbf{Шаг 1. Вспомогательное отображение $\Phi$.}

  Определим $\Phi : \Omega \to \mathbb{R}^{m+n}$:
  \[
    \Phi(x, y) \;:=\; \bigl(x,\; F(x,y)\bigr).
  \]
  Геометрический смысл: $x$-координаты не трогаем, $y$-координаты
  заменяем значением $F$. В частности, если $F(x,y) = 0$, то
  $\Phi(x,y) = (x, 0)$ --- точка на «нулевой плоскости».

  Проверяем три условия теоремы о локальном диффеоморфизме:

  \medskip
  \textit{(а) $\Phi \in C^1$.}
  Очевидно: первый блок $(x_1, \ldots, x_m)$ линеен, второй --- это $F \in C^1$.

  \medskip
  \textit{(б) $\Phi(x^0, y^0) = (x^0, 0)$.}
  Прямо из условия $F(x^0, y^0) = 0$.

  \medskip
  \textit{(в) $\mathcal{J}\Phi(x^0, y^0) \neq 0$.}
  Матрица Якоби $\Phi$ блочно-треугольная:
  \[
    \mathcal{D}\Phi(x^0, y^0)
    \;=\;
    \begin{pmatrix}
      E_{m \times m} & 0 \\[4pt]
      F'_x & F'_y
    \end{pmatrix}.
  \]
  Единичная матрица $E_{m\times m}$ стоит в левом верхнем блоке,
  потому что $\Phi$ оставляет $x$-координаты нетронутыми.
  Определитель блочно-треугольной матрицы --- произведение
  определителей диагональных блоков:
  \[
    \mathcal{J}\Phi(x^0, y^0)
    \;=\; \underbrace{\det E_{m\times m}}_{=\,1}
          \cdot \det F'_y(x^0, y^0)
    \;=\; \det F'_y(x^0, y^0) \;\neq\; 0.
  \]

  \bigskip
  \textbf{Шаг 2. Применяем теорему о локальном диффеоморфизме.}

  Якобиан $\mathcal{J}\Phi$ непрерывен (как полином от непрерывных
  аргументов), и в точке $(x^0, y^0)$ он ненулевой. По принципу
  локализации $\mathcal{J}\Phi \neq 0$ в некотором шаре вокруг $(x^0,y^0)$.
  Значит, все условия теоремы о локальном диффеоморфизме выполнены.

  Следовательно, существуют окрестности $U(x^0, y^0) \subset \Omega$
  и $V(x^0, 0)$ такие, что:
  \[
    \Phi \in C^1\!\bigl(U(x^0,y^0),\; V(x^0,0)\bigr),
    \qquad
    \Phi^{-1} \in C^1\!\bigl(V(x^0,0),\; U(x^0,y^0)\bigr).
  \]

  \bigskip
  \textbf{Шаг 3. Существование и единственность $f$.}

  Так как $V(x^0, 0)$ открыто в $\mathbb{R}^{m+n}$, существует $\delta > 0$
  такое, что $Q_\delta(x^0) \subset V(x^0, 0) \cap \mathbb{R}^m$.

  Для любой точки $x^* \in Q_\delta(x^0)$ рассмотрим точку $(x^*, 0)$
  в $V(x^0, 0)$. Так как $\Phi$ --- биекция $U(x^0,y^0)$ на $V(x^0,0)$,
  существует \emph{единственная} точка $(x^*, y^*) \in U(x^0, y^0)$ такая,
  что $\Phi(x^*, y^*) = (x^*, 0)$, то есть $F(x^*, y^*) = 0$.

  Из открытости $U(x^0, y^0)$ можно считать $y^* \in Q_\gamma(y^0)$.
  Тем самым построено отображение $f : Q_\delta(x^0) \to Q_\gamma(y^0)$,
  удовлетворяющее:
  \[
    F\bigl(x,\; f(x)\bigr) \;=\; 0
    \quad \forall\, x \in Q_\delta(x^0). \tag{$**$}
  \]

  \bigskip
  \textbf{Шаг 4. Гладкость $f \in C^1$.}

  Обратное отображение $\Phi^{-1}$ на точках вида $(x, 0)$ действует так:
  \[
    \Phi^{-1}(x, 0) \;=\; \bigl(x,\; f(x)\bigr).
  \]
  Так как $\Phi^{-1} \in C^1$, все его частные производные непрерывны.
  Значит, частные производные $f$ тоже существуют и непрерывны,
  то есть $f \in C^1$.

  \bigskip
  \textbf{Шаг 5. Формула для производной.}

  Дифференцируем тождество $(**)$ по $x$ (правило дифференцирования
  сложной функции):
  \[
    F'_x\bigl(x, f(x)\bigr)
    \;+\;
    F'_y\bigl(x, f(x)\bigr) \cdot f'_x(x)
    \;=\; 0.
  \]
  Матрица $F'_y$ обратима в окрестности $(x^0,y^0)$ по непрерывности
  и принципу локализации. Выражаем $f'_x$:
  \[
    \boxed{
      f'_x(x)
      \;=\;
      -\bigl(F'_y(x,\, f(x))\bigr)^{-1}
      \cdot F'_x(x,\, f(x)).
    }
    \qed
  \]
\end{proof}
