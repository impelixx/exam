\question{Инвариантность формы записи первого дифференциала отображения}

\begin{theorem}[Инвариантность формы записи первого дифференциала]
Пусть $f: U \subseteq \mathbb{R}^n \to \mathbb{R}^m$ дифференцируемо в точке $x^0 \in U$,
и $g: V \subseteq \mathbb{R}^m \to \mathbb{R}^p$ дифференцируемо в точке $y^0 = f(x^0) \in V$.
Тогда $h = g \circ f: U \to \mathbb{R}^p$ дифференцируема в точке $x^0$, и
\[
Dh(x^0) = Dg(f(x^0)) \cdot Df(x^0).
\]
\begin{proof}
По определению дифференцируемости $f$ в $x^0$:
\[
f(x) = f(x^0) + Df(x^0)(x - x^0) + \alpha(x),\quad \alpha(x) = o(\|x - x^0\|),
\]
для $g$ в $y^0$:
\[
g(y) = g(y^0) + Dg(y^0)(y - y^0) + \beta(y),\quad \beta(y) = o(\|y - y^0\|).
\]
Подставляем разложение $f(x)$ в $g$:
\[
h(x) = g(f(x^0)) + Dg(f(x^0))\bigl(Df(x^0)(x - x^0) + \alpha(x)\bigr)
        + \beta\bigl(Df(x^0)(x-x^0)+\alpha(x)\bigr).
\]
В силу линейности $Dg(f(x^0))$:
\[
Dg(f(x^0))(\alpha(x)) = o(\|x-x^0\|),
\]
и так как $\|Df(x^0)(x-x^0)+\alpha(x)\| \leq C\|x-x^0\|$, то
$\beta(\cdot) = o(\|x-x^0\|)$. Собирая слагаемые:
\[
h(x) = h(x^0) + \bigl(Dg(f(x^0)) Df(x^0)\bigr)(x - x^0) + o(\|x-x^0\|). \qedhere
\]
\end{proof}
\end{theorem}

\question{Признаки сравнения числовых рядов. Признаки Даламбера и Коши}

\begin{defn}[Ряд и сходимость]
  Пусть дана числовая последовательность $\{a_n\}_{n=1}^\infty$.
  Величина $S_n := \sum_{k=1}^n a_k$ называется \emph{$n$-й частичной суммой}.
  Ряд $\sum_{n=1}^\infty a_n$ \emph{сходится}, если $\exists A \in \mathbb{R}$: $\lim_{n\to\infty} S_n = A$.
\end{defn}

\begin{theorem}[Первый признак сравнения]
  Пусть $\{a_k\},\{b_k\}$ --- неотрицательные последовательности и $\exists k_0 \in \mathbb{N}$:
  $a_k \leq b_k$ при всех $k \geq k_0$.
  \begin{enumerate}
    \item $\sum b_k$ сходится $\Rightarrow$ $\sum a_k$ сходится.
    \item $\sum a_k$ расходится $\Rightarrow$ $\sum b_k$ расходится.
  \end{enumerate}
  \begin{proof}
    \textbf{(1).} Из $a_k \leq b_k$ при $k \geq k_0$ имеем
    $\sum_{k=k_0+1}^n a_k \leq \sum_{k=k_0+1}^n b_k$.
    Так как $\sum b_k$ сходится, частичные суммы ограничены, поэтому $\lim_{k\to\infty} S_n^a$ конечен.
    \textbf{(2).} Из первого пункта контрапозитивно. \qedhere
  \end{proof}
\end{theorem}

\begin{theorem}[Второй признак сравнения]
  Пусть $a_k, b_k \geq 0$, $b_k > 0$ при $k \geq k_0$, и $\exists\, L = \lim_{k\to\infty} \dfrac{a_k}{b_k} \in (0, +\infty)$.
  Тогда ряды $\sum a_k$ и $\sum b_k$ сходятся или расходятся одновременно.
  \begin{proof}
    Из $a_k/b_k \to L$: начиная с некоторого $k_1$ выполняется
    $\tfrac{L}{2}\,b_k \leq a_k \leq \tfrac{3L}{2}\,b_k$.
    По первому признаку ряды ведут себя одинаково. \qedhere
  \end{proof}
\end{theorem}

\begin{theorem}[Признак Даламбера]
  Пусть $a_k > 0$. Если $\exists k_0 \in \mathbb{N}$ и $q < 1$ такое, что
  $\dfrac{a_{k+1}}{a_k} \leq q$ при всех $k \geq k_0$, то $\sum a_k$ сходится.
  Если $\dfrac{a_{k+1}}{a_k} \geq 1$ при $k \geq k_0$, то $\sum a_k$ расходится.
  \begin{proof}
    При $\dfrac{a_{k+1}}{a_k} \leq q < 1$: начиная с $k_0$, ряд убывает и мажорируется
    геометрической прогрессией $\sum_{k \geq k_0} q^{k-k_0} a_{k_0}$, которая сходится.
    При $\dfrac{a_{k+1}}{a_k} \geq 1$: $a_k \not\to 0$ --- нарушено необходимое условие. \qedhere
  \end{proof}
\end{theorem}

\begin{theorem}[Признак Коши (радикальный)]
  Пусть $a_k \geq 0$. Если $\exists q \in (0,1)$, $\exists k_0 \in \mathbb{N}$:
  $\sqrt[k]{a_k} \leq q$ при $k \geq k_0$, то $\sum a_k$ сходится.
  Если $\sqrt[k]{a_k} \geq 1$ при $k \geq k_0$, то $\sum a_k$ расходится.
  \begin{proof}
    \textbf{Сходимость.} $a_k \leq q^k$ при $k \geq k_0$, и $\sum q^k$ сходится при $q < 1$.
    
    \textbf{Расходимость.} $a_k \not\to 0$ при $k \to \infty$ --- необходимое условие нарушено. \qedhere
  \end{proof}
\end{theorem}
