\question{Гладкие многообразия в $\mathbb{R}^m$. Эквивалентные определения гладкого многообразия}

% ═══════════════════════════════════════════════════════════════════════════
%  ОПРЕДЕЛЕНИЯ
% ═══════════════════════════════════════════════════════════════════════════

\begin{defn}[Простое многообразие]
  Пусть $1 \leq k < m$. Множество $M \subset \mathbb{R}^m$ называется
  \emph{простым $k$-мерным многообразием}, если существуют открытое
  множество $\Omega \subset \mathbb{R}^k$ и гомеоморфизм $\Phi: \Omega \to M$
  (то есть биекция такая, что $\Phi$ и $\Phi^{-1}$ непрерывны).
  Отображение $\Phi$ называется \emph{параметризацией}.
\end{defn}

\begin{defn}[Простое гладкое многообразие]
  Если дополнительно $\Phi \in C^1(\Omega, \mathbb{R}^m)$ и
  $\operatorname{rg} \mathcal{D}\Phi(x) = k$ для всех $x \in \Omega$,
  то $M$ называется \emph{простым $k$-мерным гладким многообразием}.
\end{defn}

\begin{note}
  Условие $\operatorname{rg} \mathcal{D}\Phi = k$ необходимо: без него
  $\Phi$ может занулить одно из направлений в ноль, и $k$-мерная
  область перейдёт в нечто меньшей размерности.
\end{note}

\begin{theorem}[Эквивалентные описания простого гладкого многообразия]
  \label{theorem:manifold}
  Пусть $1 \leq k < m$, $M \subset \mathbb{R}^m$, $p \in M$.
  Следующие условия эквивалентны:

  \begin{enumerate}
    \item \textbf{(Параметрически.)}
          Существует окрестность $U(p) \subset \mathbb{R}^m$ такая, что
          $U(p) \cap M$ --- простое $k$-мерное гладкое многообразие.

    \item \textbf{(Через диффеоморфизм.)}
          Существуют открытое $G \subset \mathbb{R}^m$ и диффеоморфизм
          $\Theta: G \to \mathbb{R}^m$ такие, что $p \in \Theta(G) \cap M$
          и $\Theta(G) \cap M = \Theta(G \cap \mathbb{R}^k)$.

    \item \textbf{(Через уравнения.)}
          Существуют окрестность $V \subset \mathbb{R}^m$ точки $p$ и
          функции $F_1, \ldots, F_{m-k} \in C^1(V)$ такие, что
          $\nabla F_1(p), \ldots, \nabla F_{m-k}(p)$ линейно независимы
          и
          \[
            M \cap V = \{x \in V : F_1(x) = \ldots = F_{m-k}(x) = 0\}.
          \]

    \item \textbf{(Как график.)}
          Существует окрестность $W \subset \mathbb{R}^m$ точки $p$ такая,
          что $W \cap M = \mathrm{gr}\,f$ для некоторого
          $f \in C^1(\Omega \subset \mathbb{R}^k,\, \mathbb{R}^{m-k})$
          (после перенумерации координат).
  \end{enumerate}
\end{theorem}

\begin{note}
  Доказательство идёт по кругу $1 \Rightarrow 2 \Rightarrow 3
  \Rightarrow 4 \Rightarrow 1$:
  $1\Rightarrow 2$ --- теорема о продолжении до диффеоморфизма,
  $2\Rightarrow 3$ --- свойства диффеоморфизма,
  $3\Rightarrow 4$ --- теорема о неявном отображении,
  $4\Rightarrow 1$ --- явная параметризация.
\end{note}

\begin{proof}
  \textbf{$(1 \Rightarrow 2)$: параметризация $\to$ диффеоморфизм.}

  Из условия~1 имеем параметризацию $\Phi: \widetilde{\Omega} \subset \mathbb{R}^k
  \to U(p) \cap M$ с $\Phi \in C^1$ и $\operatorname{rg}\mathcal{D}\Phi = k$.
  Это отображение из $\mathbb{R}^k$ в $\mathbb{R}^m$ с $k < m$ ---
  диффеоморфизмом быть не может, так как размерности не совпадают.

  Применяем теорему о продолжении до диффеоморфизма: «дописываем»
  недостающие $(m-k)$ координат и получаем квадратное
  $\Theta: G \subset \mathbb{R}^m \to \mathbb{R}^m$, являющееся
  $C^1$-диффеоморфизмом, причём $\Theta\big|_{G \cap \mathbb{R}^k} = \Phi$.
  Тогда:
  \[
    p \in \Theta(G) \cap M
    \quad\text{и}\quad
    \Theta(G \cap \mathbb{R}^k) = \Theta(G) \cap M.
  \]

  \medskip
  \textbf{$(2 \Rightarrow 3)$: диффеоморфизм $\to$ уравнения.}

  Диффеоморфизм $\Theta$ «выпрямляет» $M$ в плоскость $\mathbb{R}^k$,
  а $\Theta^{-1}$ переводит эту плоскость обратно в $M$.
  Положим $V := \Theta(G)$ и определим $F_1, \ldots, F_{m-k}$
  как последние $(m-k)$ компонент $\Theta^{-1}: V \to G$.
  Они $C^1$-гладкие, так как $\Theta^{-1} \in C^1$.

  На $M \cap V$ функции $F_i$ равны нулю: точки $M$ переходят через
  $\Theta^{-1}$ в $G \cap \mathbb{R}^k$, где последние $(m-k)$ координат
  нулевые. Следовательно:
  \[
    M \cap V = \{x \in V : F_1(x) = \ldots = F_{m-k}(x) = 0\}.
  \]
  Линейная независимость градиентов: $\mathcal{J}\Theta^{-1} \neq 0$
  на $V$ означает, что все строки $\mathcal{D}\Theta^{-1}(p)$ линейно
  независимы. Строки матрицы Якоби --- это градиенты координатных
  функций, поэтому $\nabla F_1(p), \ldots, \nabla F_{m-k}(p)$
  линейно независимы.

  \medskip
  \textbf{$(3 \Rightarrow 4)$: уравнения $\to$ график.}

  Определим $F := (F_1, \ldots, F_{m-k}) \in C^1(V, \mathbb{R}^{m-k})$.
  Тогда $F(p) = 0$, а градиенты $\nabla F_i(p)$ линейно независимы,
  значит $\operatorname{rg}\mathcal{D}F(p) = m-k$.
  После перенумерации координат последние $(m-k)$ столбцов образуют
  матрицу $F'_y(p)$ с $\det F'_y(p) \neq 0$ --- это ровно условие
  теоремы о неявном отображении. Применяем её: существуют $\delta,
  \gamma > 0$ и $f \in C^1(Q_\delta(p_x), Q_\gamma(p_y))$ такое,
  что $F(x, f(x)) = 0$. Полагая $W := Q_\delta(p_x) \times Q_\gamma(p_y)$,
  получаем $W \cap M = \mathrm{gr}\,f$.

  \medskip
  \textbf{$(4 \Rightarrow 1)$: график $\to$ параметризация.}

  Самый простой переход. Определим $\Phi: \Omega \to W \cap M$:
  \[
    \Phi(x) := \bigl(x,\; f(x)\bigr).
  \]
  \textit{Биективность:} $x_1 \neq x_2 \Rightarrow
  (x_1, f(x_1)) \neq (x_2, f(x_2))$.

  \textit{Непрерывность $\Phi$:} следует из $f \in C^1$.

  \textit{Непрерывность $\Phi^{-1}$:} обратное $\Phi^{-1}(y) =
  y \cap \mathbb{R}^k$ --- проекция на первые $k$ координат,
  непрерывна явно.

  Значит $\Phi$ --- гомеоморфизм. Матрица Якоби:
  \[
    \mathcal{D}\Phi(x) =
    \begin{pmatrix} E_{k\times k} \\ \mathcal{D}f(x) \end{pmatrix}.
  \]
  Первые $k$ строк --- единичная матрица, значит
  $\operatorname{rg}\mathcal{D}\Phi = k$.
  Следовательно, $W \cap M$ --- простое $k$-мерное гладкое
  многообразие, то есть условие~1 выполнено с $U(p) := W$. \qed
\end{proof}
