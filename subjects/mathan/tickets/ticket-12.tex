\question{Формула Тейлора с остаточным членом в форме Пеано и в форме Лагранжа для ФНП}

\begin{note}
Идея: вводим $\varphi(t) = F(x_* + t\,dx)$ и сводим задачу к одномерному Тейлору.
\end{note}

\begin{theorem}[Остаток Лагранжа]
Пусть все частные производные $F$ до порядка $m+1$ существуют в $B_\delta(x_*)$.
Тогда $\forall x \in B_\delta(x_*)$, $dx = x - x_*$:
\[
    F(x) = \sum_{k=0}^{m} \frac{d^k_{x_*} F(dx)}{k!}
    + \frac{d^{m+1}_{x^\theta} F(dx)}{(m+1)!},
    \quad \theta \in (0,1),
    \quad x^\theta = x_* + \theta\,dx.
\]
\end{theorem}

\begin{proof}
Введём $\varphi(t) = F(x_* + t\,dx)$, тогда $\varphi(0) = F(x_*)$, $\varphi(1) = F(x)$.
По правилу дифференцирования композиции $\varphi'(t) = DF(x^t)\cdot dx$,
откуда по индукции $\varphi^{(k)}(t) = d^k_{x^t}F(dx)$, $x^t = x_*+t\,dx$.
В частности $\varphi^{(k)}(0) = d^k_{x_*}F(dx)$.
Применяем одномерный Тейлор к $\varphi$ на $[0,1]$. \qedhere
\end{proof}

\begin{theorem}[Остаток Пеано]
Пусть все производные $F$ до порядка $m$ существуют и непрерывны в $B_\delta(x_*)$.
Тогда:
\[
    F(x) = \sum_{k=0}^{m} \frac{d^k_{x_*} F(dx)}{k!}
    + o\!\left(\|x - x_*\|^m\right), \quad x \to x_*.
\]
\end{theorem}

\begin{proof}
Применяем Лагранжа с $m-1$, добавляем и вычитаем $\frac{d^m_{x_*}F(dx)}{m!}$:
\[
    F(x) = \sum_{k=0}^{m} \frac{d^k_{x_*} F(dx)}{k!}
    + \frac{d^m_{x^\theta}F(dx) - d^m_{x_*}F(dx)}{m!}.
\]
Остаток $R$: из непрерывности производных $m$-го порядка
\[
    |R| \leq \frac{n^m\,\varepsilon(x)}{m!}\cdot\|x-x_*\|^m = o\!\left(\|x-x_*\|^m\right). \qedhere
\]
\end{proof}

\question{Интегральный признак сходимости числового ряда}

\begin{theorem}[Признак Коши--Маклорена]
  Пусть $f: [1, +\infty) \to [0, +\infty)$ --- монотонно убывающая функция. Тогда
  \[
    \sum_{k=1}^{\infty} f(k) \text{ сходится} \quad\Longleftrightarrow\quad
    \int_1^{+\infty} f(x)\,dx \text{ сходится}.
  \]
  \begin{proof}
    \textbf{Случай $A \neq 0$.}
    В силу монотонности $\exists\lim_{x\to\infty} f(x) = A \in \mathbb{R}$.
    Если $A \neq 0$, то $\lim_{k\to\infty} f(k) = A \neq 0$ --- нарушено необходимое условие
    сходимости ряда $\Rightarrow$ ряд расходится; и $\int_1^\infty f(x)\,dx$ расходится
    по признаку сравнения. Оба расходятся.

    \textbf{Случай $A = 0$.}
    Так как $f$ убывает, при $x \in [k, k+1]$ имеем $f(k) \geq f(x) \geq f(k+1)$.
    Интегрируя по $[k, k+1]$:
    \[
      f(k) \geq \int_k^{k+1} f(x)\,dx \geq f(k+1).
    \]
    Суммируя по $k = 1, \ldots, n$:
    \[
      \sum_{k=1}^n f(k) \geq \int_1^{n+1} f(x)\,dx \geq \sum_{k=2}^{n+1} f(k).
    \]
    Следовательно, частичные суммы ряда ограничены тогда и только тогда, когда
    интеграл ограничен, т.\,е.\ ряд и интеграл сходятся или расходятся одновременно. \qedhere
  \end{proof}
\end{theorem}
