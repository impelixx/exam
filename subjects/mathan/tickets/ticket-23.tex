\question{Геометрический смысл дифференцируемости и градиента}

\begin{defn}[Градиент]
  Пусть $f : \mathbb{R}^n \to \mathbb{R}$, $x^0 \in \mathbb{R}^n$.
  Градиентом функции $f$ в точке $x^0$ называется вектор частных производных:
  \[
    \nabla f(x^0)
    = \left(\frac{\partial f}{\partial x_1}(x^0), \ldots,
             \frac{\partial f}{\partial x_n}(x^0)\right),
  \]
  где $\dfrac{\partial f}{\partial x_k}(x^0)
       = \lim_{t \to 0}\dfrac{f(x^0 + t\,e_k) - f(x^0)}{t}$,
  а $e_k = (0,\ldots,\underset{k}{1},\ldots,0)$ --- орт $k$-й оси.
\end{defn}

\begin{defn}[Касательная плоскость]
  Пусть $f : B_\delta(x^0, y^0) \to \mathbb{R}$. Плоскость вида
  \[
    z_t(x,y) = f(x^0,y^0) + N_x(x-x^0) + N_y(y-y^0)
  \]
  называется \textit{касательной плоскостью} к графику $f$ в точке
  $(x^0,\, y^0,\, f(x^0,y^0))$, если:
  \[
    f(x,y) - z_t(x,y)
    = \varepsilon(x,y)\cdot\sqrt{(x-x^0)^2+(y-y^0)^2},
  \]
  где $\varepsilon(x,y) \to 0$ при $(x,y) \to (x^0, y^0)$.

  Геометрически: плоскость задаётся точкой касания
  $(x^0, y^0, f(x^0,y^0))$ и двумя наклонами ---
  $\frac{\partial f}{\partial x}$ вдоль оси $x$
  и $\frac{\partial f}{\partial y}$ вдоль оси $y$.
  Нормальный вектор к такой плоскости:
  $\overline{N} = \!\left(\dfrac{\partial f}{\partial x}(x^0,y^0),\;
  \dfrac{\partial f}{\partial y}(x^0,y^0),\; -1\right)$.
\end{defn}

\begin{theorem}[Геометрический смысл дифференцируемости]
  $f : B_\delta(x^0,y^0) \to \mathbb{R}$ дифференцируема в точке $(x^0,y^0)$
  тогда и только тогда, когда существует касательная плоскость к графику $f$
  в точке $(x^0,\,y^0,\,f(x^0,y^0))$. При этом:
  \[
    \overline{n} = \bigl(\nabla f(x^0,y^0),\,-1\bigr),
  \]
  а дифференциал --- это приращение $z$ вдоль касательной плоскости:
  \[
    z_t(x,y) - z_t(x^0,y^0) = d_{(x^0,y^0)}f\bigl((x,y)-(x^0,y^0)\bigr).
  \]
  \begin{proof}
    Дифференцируемость $f$ в $(x^0,y^0)$ равносильна существованию
    $A_1, A_2 \in \mathbb{R}$ таких, что:
    \[
      f(x,y) - f(x^0,y^0)
      = A_1(x-x^0) + A_2(y-y^0)
      + \varepsilon(x,y)\sqrt{(x-x^0)^2+(y-y^0)^2}.
    \]
    Плоскость $z_t(x,y) = f(x^0,y^0) + A_1(x-x^0) + A_2(y-y^0)$
    удовлетворяет определению касательной, так как
    $f(x,y) - z_t(x,y) = \varepsilon(x,y)\sqrt{(x-x^0)^2+(y-y^0)^2}$.
    Обратно --- существование касательной даёт то же равенство,
    то есть определение дифференцируемости.
    Поскольку $(A_1,A_2) = \nabla f(x^0,y^0)$,
    нормаль есть $(\nabla f(x^0,y^0),\,-1)$. \qedhere
  \end{proof}
\end{theorem}
