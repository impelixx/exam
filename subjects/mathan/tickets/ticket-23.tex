\question{Производная по направлению и градиент. Геометрический смысл градиента}

% NOTE: ticket-06 Q1 уже содержит геометрический смысл касательной плоскости.
% Здесь сосредоточимся на производной по направлению и связи с градиентом.

\begin{defn}[Производная по направлению]
  Пусть $f: \mathbb{R}^n \to \mathbb{R}$, $x^0 \in \mathbb{R}^n$, $l \in \mathbb{R}^n$, $\|l\| = 1$.
  \emph{Производной $f$ по направлению $l$ в точке $x^0$} называется
  \[
    \frac{\partial f}{\partial l}(x^0) := \lim_{t \to +0} \frac{f(x^0 + tl) - f(x^0)}{t},
  \]
  если предел существует.
\end{defn}

\begin{theorem}[Связь с градиентом]
  Если $f$ дифференцируема в точке $x^0$, то для любого единичного вектора $l$:
  \[
    \frac{\partial f}{\partial l}(x^0) = \langle \nabla f(x^0),\, l \rangle.
  \]
  \begin{proof}
    Подставим $x = x^0 + tl$ в определение дифференцируемости:
    \[
      f(x^0+tl) - f(x^0) = \langle \nabla f(x^0), tl \rangle + o(t\|l\|)
      = t\langle \nabla f(x^0), l \rangle + o(t).
    \]
    Делим на $t > 0$ и переходим к пределу. \qedhere
  \end{proof}
\end{theorem}

\begin{theorem}[Геометрический смысл градиента]
  Если $\nabla f(x^0) \neq 0$, то:
  \begin{enumerate}
    \item Направление наибыстрейшего роста $f$ --- это $l_{\max} = \dfrac{\nabla f(x^0)}{\|\nabla f(x^0)\|}$.
    \item Максимальное значение производной по направлению: $\max_{\|l\|=1} \dfrac{\partial f}{\partial l} = \|\nabla f(x^0)\|$.
    \item Градиент перпендикулярен линиям уровня $\{x: f(x) = c\}$.
  \end{enumerate}
  \begin{proof}
    По неравенству Коши--Буняковского:
    $\langle \nabla f, l \rangle \leq \|\nabla f\|\cdot\|l\| = \|\nabla f\|$,
    и равенство достигается при $l = \nabla f/\|\nabla f\|$. \qedhere
  \end{proof}
\end{theorem}

\question{Понятие аналитичности в точке. Достаточное условие аналитичности. Пример неаналитичной функции}

\begin{defn}[Аналитическая функция]
  Функция $f: (a,b) \to \mathbb{R}$ называется \emph{аналитической в точке $x_0 \in (a,b)$},
  если существует $r > 0$ и степенной ряд $\sum_{n=0}^\infty c_n(x-x_0)^n$,
  сходящийся к $f(x)$ на $(x_0-r, x_0+r)$.
\end{defn}

\begin{theorem}[Достаточное условие аналитичности]
  Если $f \in C^\infty(x_0-r, x_0+r)$ и существует $M > 0$ такое, что
  \[
    |f^{(n)}(x)| \leq M^n \quad\forall n \in \mathbb{N},\; \forall x \in (x_0-r, x_0+r),
  \]
  то $f$ аналитична в $x_0$.
  \begin{proof}
    Оценим остаток Лагранжа:
    $|R_n(x)| = \left|\dfrac{f^{(n+1)}(\xi)}{(n+1)!}(x-x_0)^{n+1}\right|
    \leq \dfrac{M^{n+1}|x-x_0|^{n+1}}{(n+1)!} \to 0$
    при $n\to\infty$ для любого фиксированного $x$ (факториал растёт быстрее показательной). \qedhere
  \end{proof}
\end{theorem}

\begin{example}[Неаналитичная бесконечно дифференцируемая функция]
  \[
    f(x) = \begin{cases} e^{-1/x^2}, & x \neq 0, \\ 0, & x = 0. \end{cases}
  \]
  Можно показать, что $f \in C^\infty(\mathbb{R})$ и $f^{(n)}(0) = 0$ для всех $n$.
  Значит, ряд Тейлора в нуле тождественно равен нулю, но $f(x) \neq 0$ при $x \neq 0$.
  Следовательно, $f$ не аналитична в точке $x = 0$.
\end{example}
