\question{Теорема об открытом отображении}

\begin{defn}[Открытое множество]
  $A \subset \mathbb{R}^m$ называется \emph{открытым}, если $\forall x_0 \in A\;\exists r>0: U_r(x_0) \subset A$.
\end{defn}

\begin{defn}[Дифференцируемое отображение]
  $F: \Omega \to \mathbb{R}^m$ \emph{дифференцируемо в точке $x_0$}, если $\exists$ линейное
  $DF(x_0): \mathbb{R}^n \to \mathbb{R}^m$ такое, что
  $F(x) = F(x_0) + DF(x_0)(x-x_0) + o(\|x-x_0\|)$.
\end{defn}

\begin{lemma}[Обратимый оператор растягивает на константу]
  Пусть $A \in \mathcal{L}(\mathbb{R}^n, \mathbb{R}^n)$ обратима. Тогда
  $\exists c > 0: \forall x,y \in \mathbb{R}^n$
  \[
    \|A(x) - A(y)\| \geq c\,\|x - y\|, \quad c = \frac{1}{\|A^{-1}\|}.
  \]
  \begin{proof}
    $\|x\| = \|A^{-1}(A(x))\| \leq \|A^{-1}\|\cdot\|A(x)\|$, откуда
    $\|A(x)\| \geq \tfrac{1}{\|A^{-1}\|}\,\|x\|$.
    Подставляя $x - y$: $\|A(x-y)\| \geq \tfrac{1}{\|A^{-1}\|}\,\|x-y\|$. \qedhere
  \end{proof}
\end{lemma}

\begin{theorem}[Об открытом отображении]
  Пусть $\Omega \subset \mathbb{R}^n$ открыто, $F \in \mathrm{DIF}(\Omega, \mathbb{R}^n)$
  и $\mathcal{J}_F(x) \neq 0$ для всех $x \in \Omega$. Тогда $F(\Omega)$ открыто в $\mathbb{R}^n$.
  \begin{proof}
    Зафиксируем $x^0 \in \Omega$, $y^0 = F(x^0)$. По основной геометрической теореме
    $\exists\delta > 0,\, c > 0: \|F(x)-F(x^0)\| \geq c\|x-x^0\|$
    для всех $x \in B_\delta(x^0)$.

    Покажем, что $B_{c\delta/2}(y^0) \subset F(\Omega)$.
    Для $y^* \in B_{c\delta/2}(y^0)$ рассмотрим $h(x) = \|F(x)-y^*\|$ на $\overline{B_\delta(x^0)}$.

    \begin{itemize}
      \item В центре: $h(x^0) = \|y^0 - y^*\| < \tfrac{c\delta}{2}$.
      \item На сфере: $h(x) \geq c\delta - \|y^0-y^*\| > \tfrac{c\delta}{2}$.
    \end{itemize}

    Минимум $h$ достигается на компакте в interior точке $x^* \in B_\delta(x^0)$.
    В этой точке $\nabla\|F-y^*\|^2 = 0$, т.\,е.\ $(\mathcal{D}F(x^*))^T (F(x^*)-y^*) = 0$.
    Так как $\mathcal{J}_F(x^*) \neq 0$, матрица $\mathcal{D}F(x^*)$ обратима $\Rightarrow F(x^*) = y^*$. \qedhere
  \end{proof}
\end{theorem}

\question{Свойства интеграла с переменным верхним пределом. Формула Ньютона--Лейбница}

\begin{defn}[Интеграл с переменным верхним пределом]
  Если $f$ интегрируема на $[a,b]$, то $\forall x \in [a,b]$ существует
  \[
    F(x) = \int_a^x f(t)\,dt,
  \]
  называемый \emph{интегралом с переменным верхним пределом}.
\end{defn}

\begin{theorem}[Непрерывность]
  Если $f$ интегрируема на $[a,b]$, то $F(x)$ непрерывна на $[a,b]$.
  \begin{proof}
    Пусть $x, x+\Delta x \in [a,b]$. По аддитивности:
    \[
      \Delta F = F(x+\Delta x) - F(x) = \int_x^{x+\Delta x} f(t)\,dt.
    \]
    Так как $f$ интегрируема, то ограничена: $|f(t)| \leq M$.
    Поэтому $|\Delta F| \leq M|\Delta x| \to 0$ при $\Delta x \to 0$. \qedhere
  \end{proof}
\end{theorem}

\begin{theorem}[Дифференцируемость. Теорема Барроу]
  Если $f$ интегрируема на $[a,b]$ и непрерывна в точке $x_0$, то
  $F(x) = \int_a^x f(t)\,dt$ дифференцируема в $x_0$ и $F'(x_0) = f(x_0)$.
  \begin{proof}
    \[
      \left|\frac{F(x) - F(x_0)}{x - x_0} - f(x_0)\right|
      = \left|\frac{1}{x-x_0}\int_{x_0}^{x}(f(t) - f(x_0))\,dt\right|
      \leq \frac{1}{|x-x_0|}\int_{x_0}^x |f(t)-f(x_0)|\,dt.
    \]
    По непрерывности $f$ в $x_0$: $\forall\varepsilon>0\;\exists\delta>0$ такое, что
    $|f(t)-f(x_0)| < \varepsilon$ при $|t-x_0|<\delta$.
    При $|x-x_0|<\delta$ правая часть $< \varepsilon$. \qedhere
  \end{proof}
\end{theorem}

\begin{theorem}[Существование первообразной у непрерывной функции]
  Если $f$ непрерывна на $[a,b]$, то она имеет первообразную на этом отрезке,
  и ею является $\int_a^x f(t)\,dt + C$.
  \begin{proof}
    Берём $x \in [a,b]$. По теореме Барроу $F'(x) = f(x)$.
    Так как $x$ произвольно и $f$ непрерывна, $F$ --- первообразная. \qedhere
  \end{proof}
\end{theorem}

\begin{theorem}[Формула Ньютона--Лейбница]
  Если $f$ непрерывна на $[a,b]$ и $\Phi$ --- любая первообразная $f$, то
  \[
    \int_a^b f(x)\,dx = \Phi(b) - \Phi(a).
  \]
  \begin{proof}
    Пусть $\Phi$ --- первообразная $f$. Тогда $F(x) = \Phi(x) + C$.
    Из $F(a) = \int_a^a f = 0$ получаем $C = -\Phi(a)$, откуда
    \[
      \int_a^b f(t)\,dt = F(b) = \Phi(b) + C = \Phi(b) - \Phi(a). \qedhere
    \]
  \end{proof}
\end{theorem}
