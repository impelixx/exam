\question{Теорема об открытом отображении}

\begin{defn}[Линейный оператор]
  Пусть у нас есть $\R^n$ и $\R^m$. Линейным оператором $A : \R^n \to \R^m$ называется отображение, для которого выполняются следующие условия:
\begin{enumerate}
  \item $A(x+y) = Ax + Ay$ для всех $x,y \in \R^n$;
  \item $A(\alpha x) = \alpha Ax$ для всех $x \in \R^n$ и $\alpha \in \R$.
\end{enumerate}
\end{defn}


\begin{lemma}[линейный оператор, растягивает минимум на какую то константу]
  Пусть $A : \R^n \to \R^n$ --- линейный оператор и матрица обратима, тогда для всех $x,y \in \R^n$ выполняется неравенство
  \[
  \|A(x) - A(y)\| \geq \frac{1}{\|A^{-1}\|}\|x-y\|
  \]
  \begin{proof}
  Выберем $x \in \R^n$. Так как у нас есть путь туда и обратно, то можно записать $x = A^{-1}(A(x))$. Теперь, если мы выберем $y \in \R^n$, то можно заюзать прикол про рандомное неравенство и переписать вот так: $\|x\| = \|A^{-1}(A(x))\| \leq \|A^{-1}\|\|A(x)\|$ и при этом у нас $\|A^{-1}\| \neq 0$, то можно поделить и сказать, что $\|A(x)\| \geq \frac{\|x\|}{\|A^{-1}\|}$. А теперь выдаем что оказывает под $x$ все это время было $x - y$, и получаем искомое неравенство.
  \end{proof}
\end{lemma}
Из леммы выше несложно заметить, что есть магическая константа $M = \frac{1}{\|A^{-1}\|}$, с минимальным растяжением, которая не зависит от выбора $x$ и $y$. но можно более нормально сформулировать
\begin{theorem}[Основная геом теорема]
  Пусть $\Omega \subset \R^n, F : \Omega \to \R^n$ и $x^0$ --- внутренняя точка $\Omega$. Если $F$ дифференцируема в $x^0$ и матрица Якоби $DF(x^0)$ обратима, то найдется окрестность $U$ точки $x^0$, такая что для всех $x \in U$ выполняется неравенство $\|F(x) - F(x^0)\| \geq M\|x-x^0\|$, где $M > 0$ --- константа, которая не зависит от выбора $x$.
  \begin{proof}
 так как $x^0$ --- точка внутри $\Omega$,  и она дифференцируем в точке $x^0$, то можно по определению \[ F(x) = F(x^0) + DF(x^0)(x-x^0) + \varepsilon(x)\|x-x^0\|, \] где $\varepsilon(x) \to 0$ при $x \to x^0$. 
 Теперь вспомним, что \[ \|DF(x^0)\|\|x-x^0\| \geq \frac{1}{\|DF(x^0)^{-1}\|}\|x-x^0\| \], потом, так как $\varepsilon(x) \to 0$ при $x \to x^0$, то справедливо неравенство $\|\varepsilon(x)\|\|x-x^0\| \leq \frac{1}{2\|DF(x^0)^{-1}\|}\|x-x^0\|$ и теперь, 
 вспомнив неравенство треугольника $|a + b| \geq |a| - |b|$,
 
 можно записать \[ \|F(x) - F(x^0)\| \geq \|DF(x^0)(x-x^0)\| - \|\varepsilon(x)\|\|x-x^0\| \geq \frac{1}{2\|DF(x^0)^{-1}\|}\|x-x^0\|. \].
  
\end{proof} 
\end{theorem}
\begin{theorem}[Теорема об открытом отображении]
Пусть $\Omega \subset \R^n$ --- открытое множество и $F \in \mathrm{DIF}(\Omega, \R^n)$.
Пусть $\mathcal{J}_F(x) \neq 0$ для всех $x \in \Omega$.
Тогда $F(\Omega)$ открыто в $\R^n$.
\begin{proof}
Зафиксируем произвольную точку $x^0 \in \Omega$ и положим $y^0 = F(x^0)$.
Достаточно найти открытый шар с центром $y^0$, лежащий в $F(\Omega)$.

\textbf{Шаг 1.} По основной геометрической теореме (все её условия выполнены):
\[
  \exists\,\delta > 0 \;\exists\, c > 0 \;:\;\;
  B_\delta(x^0) \subset \Omega
  \;\;\text{и}\;\;
  \|F(x) - F(x^0)\| \geq c\|x - x^0\|
  \quad \forall x \in B_\delta(x^0).
\]

\textbf{Шаг 2.} Покажем, что $B_{c\delta/2}(y^0) \subset F(\Omega)$.
Возьмём произвольное $y^* \in B_{c\delta/2}(y^0)$, то есть
$\|y^0 - y^*\| < \dfrac{c\delta}{2}$.
Нужно найти $x^* \in \Omega$ такое, что $F(x^*) = y^*$.

\textbf{Шаг 3.} Рассмотрим функцию $h(x) = \|F(x) - y^*\|$ на замкнутом шаре
$\overline{B_\delta(x^0)}$. Оценим её в двух местах:
\begin{itemize}
  \item В центре: $h(x^0) = \|y^0 - y^*\| < \dfrac{c\delta}{2}$.
  \item На сфере $S_\delta(x^0)$: по неравенству треугольника и шагу 1,
  \[
    h(x) \geq \|F(x)-y^0\| - \|y^0-y^*\|
    \geq c\|x-x^0\| - \|y^0-y^*\|
    = c\delta - \|y^0-y^*\|
    > c\delta - \frac{c\delta}{2} = \frac{c\delta}{2}.
  \]
\end{itemize}

\textbf{Шаг 4.} Функция $h$ непрерывна, а $\overline{B_\delta(x^0)}$ компактна.
По теореме Вейерштрасса $h$ достигает минимума.
Из оценок шага 3: на границе $h > c\delta/2$, а в центре $h < c\delta/2$,
поэтому минимум достигается в некоторой внутренней точке $x^* \in B_\delta(x^0)$.

\textbf{Шаг 5.} В точке внутреннего минимума $x^*$ необходимое условие:
$\nabla h^2(x^*) = 0$. Раскрывая $h^2(x) = \sum_{j=1}^n (F_j(x) - y^*_j)^2$:
\[
  \sum_{j=1}^n 2(F_j(x^*) - y^*_j)\,\frac{\partial F_j}{\partial x_i}(x^*) = 0
  \quad \forall\, i = 1,\ldots,n.
\]
В матричной форме (обозначим $z = F(x^*) - y^*$):
\[
  (\mathcal{D}F(x^*))^T z = 0.
\]
Так как $\mathcal{J}_F(x^*) \neq 0$, матрица $\mathcal{D}F(x^*)$ обратима,
значит $z = 0$, то есть $F(x^*) = y^*$.

Точка $y^*$ выбиралась произвольно из $B_{c\delta/2}(y^0)$,
поэтому $B_{c\delta/2}(y^0) \subset F(\Omega)$.
Так как $x^0$ выбирался произвольно, $F(\Omega)$ открыто.
\end{proof}
\end{theorem}
