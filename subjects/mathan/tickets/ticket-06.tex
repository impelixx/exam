\question{Геометрический смысл дифференцируемости и градиента}

\begin{defn}[Градиент]
  Пусть $f : \mathbb{R}^n \to \mathbb{R}$, $x^0 \in \mathbb{R}^n$.
  Градиентом функции $f$ в точке $x^0$ называется вектор частных производных:
  \[
    \nabla f(x^0)
    = \left(\frac{\partial f}{\partial x_1}(x^0), \ldots,
             \frac{\partial f}{\partial x_n}(x^0)\right).
  \]
\end{defn}

\begin{defn}[Касательная плоскость]
  Плоскость $z_t(x,y) = f(x^0,y^0) + N_x(x-x^0) + N_y(y-y^0)$
  называется \emph{касательной плоскостью} к графику $f$ в точке $(x^0, y^0, f(x^0,y^0))$, если
  \[
    f(x,y) - z_t(x,y) = \varepsilon(x,y)\cdot\sqrt{(x-x^0)^2+(y-y^0)^2},
    \quad \varepsilon(x,y) \to 0 \text{ при } (x,y) \to (x^0,y^0).
  \]
  Нормальный вектор: $\overline{N} = \bigl(\tfrac{\partial f}{\partial x}(x^0,y^0),\;
  \tfrac{\partial f}{\partial y}(x^0,y^0),\; -1\bigr)$.
\end{defn}

\begin{theorem}[Геометрический смысл дифференцируемости]
  $f$ дифференцируема в $(x^0,y^0)$ тогда и только тогда, когда существует
  касательная плоскость к её графику в точке $(x^0, y^0, f(x^0,y^0))$.
  При этом нормаль $\overline{n} = (\nabla f(x^0,y^0),\,-1)$,
  а дифференциал $df$ есть приращение $z$ вдоль касательной:
  \[
    z_t(x,y) - z_t(x^0,y^0) = d_{(x^0,y^0)}f\bigl((x,y)-(x^0,y^0)\bigr).
  \]
  \begin{proof}
    Дифференцируемость равносильна $f(x,y) - f(x^0,y^0) = A_1(x-x^0) + A_2(y-y^0)
    + \varepsilon\sqrt{(x-x^0)^2+(y-y^0)^2}$. Плоскость
    $z_t = f(x^0,y^0) + A_1(x-x^0) + A_2(y-y^0)$ удовлетворяет определению.
    Обратно --- существование касательной даёт то же равенство, где $(A_1,A_2) = \nabla f(x^0,y^0)$. \qedhere
  \end{proof}
\end{theorem}

\paragraph{Производная по направлению.}
Пусть $l = (l_1, l_2, l_3)$ --- вектор в $\mathbb{R}^3$.
Производной $f$ по вектору $l$ в точке $(x_0,y_0,z_0)$ называется
\[
  \frac{\partial f}{\partial l} = \lim_{t \to +0}
  \frac{f(x_0+tl_1,\,y_0+tl_2,\,z_0+tl_3) - f(x_0,y_0,z_0)}{t}.
\]
Нетрудно заметить, что если $f$ дифференцируема:
\[
  \frac{\partial f}{\partial l}(x_0,y_0,z_0)
  = \frac{\partial f}{\partial x}(x_0,y_0,z_0)\,l_1
  + \frac{\partial f}{\partial y}(x_0,y_0,z_0)\,l_2
  + \frac{\partial f}{\partial z}(x_0,y_0,z_0)\,l_3
  = \langle l,\, \nabla f \rangle.
\]
Таким образом, $\dfrac{\partial f}{\partial l} = \langle l, \nabla f \rangle$, и при $l = \nabla f$ производная по направлению максимальна.

\question{Разложение функции $e^z$ в ряд Тейлора (Маклорена)}

\begin{defn}[Ряд Тейлора]
  Если $f(x)$ определена в $x_0$ и имеет там все производные, то степенной ряд
  \[
    f(x_0) + \sum_{n=1}^{\infty} \frac{f^{(n)}(x_0)}{n!}(x-x_0)^n
  \]
  называется \emph{рядом Тейлора} функции $f$ в точке $x_0$.
\end{defn}

\begin{defn}[Регулярность / аналитичность]
  Функция $f$ называется \emph{регулярной (аналитической) в точке $x_0$},
  если она представима сходящимся степенным рядом:
  $f(x) = \sum_{n=0}^{\infty} c_n(x-x_0)^n$.
  Коэффициенты определяются формулой $c_n = \dfrac{f^{(n)}(x_0)}{n!}$.
\end{defn}

\begin{theorem}[Разложение $e^x$ в ряд Маклорена]
  \[
    e^x = 1 + \sum_{n=1}^{\infty} \frac{x^n}{n!} = \sum_{n=0}^{\infty} \frac{x^n}{n!},
    \quad x \in \mathbb{R},\quad R = +\infty.
  \]
  \begin{proof}
    Так как $f^{(n)}(x) = e^x$ для всех $n$, то $c_n = \dfrac{f^{(n)}(0)}{n!} = \dfrac{1}{n!}$.
    Оценим радиус сходимости: для любого фиксированного $\rho \geq 0$ и $\forall x \in (-\rho, \rho)$
    выполняется $0 \leq f^{(n)}(x) = e^x \leq e^\rho$, поэтому остаток Лагранжа
    \[
      R_n(x) = \frac{e^\theta}{(n+1)!}x^{n+1} \to 0 \text{ при } n \to \infty
    \]
    для любого фиксированного $x$. Поскольку $\rho \in \mathbb{N}$ произвольно, $R = +\infty$.
  \end{proof}
\end{theorem}

\begin{note}
  Разложение обобщается на комплексные числа: $e^z = \sum_{n=0}^{\infty} \dfrac{z^n}{n!}$,
  $z \in \mathbb{C}$, что используется в доказательстве формулы Эйлера $e^{i\varphi} = \cos\varphi + i\sin\varphi$.
\end{note}
