\question{Геометрический смысл дифференцируемости и градиента}

\begin{defn}[Градиент]
  Пусть $f : \mathbb{R}^n \to \mathbb{R}$, $x^0 \in \mathbb{R}^n$.
  Градиентом функции $f$ в точке $x^0$ называется вектор частных производных:
  \[
    \nabla f(x^0)
    = \left(\frac{\partial f}{\partial x_1}(x^0), \ldots,
             \frac{\partial f}{\partial x_n}(x^0)\right),
  \]
  где $\dfrac{\partial f}{\partial x_k}(x^0)
       = \lim_{t \to 0}\dfrac{f(x^0 + t\,e_k) - f(x^0)}{t}$,
  а $e_k = (0,\ldots,\underset{k}{1},\ldots,0)$ --- орт $k$-й оси.
\end{defn}

\begin{defn}[Касательная плоскость]
  Плоскость вида
  \[
    z_t(x, y)
    = f(x^0, y^0) + N_x(x - x^0) + N_y(y - y^0)
  \]
  называется \textit{касательной} к графику $f$ в точке
  $(x^0,\, y^0,\, f(x^0, y^0))$, если
  \[
    f(x,y) - z_t(x,y)
    = o\!\left(\sqrt{(x-x^0)^2+(y-y^0)^2}\right),
    \quad (x,y)\to(x^0,y^0).
  \]
\end{defn}

\begin{theorem}[Геометрический смысл дифференцируемости]
  Функция $f : \mathbb{R}^2 \to \mathbb{R}$ дифференцируема в точке
  $(x^0, y^0)$ тогда и только тогда, когда существует касательная
  плоскость к графику $f$ в точке $(x^0,\, y^0,\, f(x^0, y^0))$.
  При этом нормальный вектор к касательной плоскости равен
  $\mathbf{n} = (\nabla f(x^0, y^0),\, -1)$, а дифференциал ---
  это приращение $z$ вдоль касательной плоскости:
  \[
    df(x^0)(h) = z_t(x^0 + h) - z_t(x^0).
  \]
  \begin{proof}
    Дифференцируемость в точке $(x^0, y^0)$ означает существование
    $A_1, A_2 \in \mathbb{R}$ таких, что:
    \[
      f(x,y) - f(x^0,y^0)
      = A_1(x-x^0) + A_2(y-y^0)
        + o\!\left(\sqrt{(x-x^0)^2+(y-y^0)^2}\right).
    \]
    Это в точности определение касательной плоскости с
    $N_x = A_1$, $N_y = A_2$.
    Обратно --- если касательная плоскость существует, то то же
    равенство даёт определение дифференцируемости.
    Поскольку $(A_1, A_2) = \nabla f(x^0, y^0)$, нормальный вектор
    есть $(\nabla f(x^0, y^0),\,-1)$. \qedhere
  \end{proof}
\end{theorem}

\begin{theorem}[Геометрический смысл градиента]
  Пусть $f$ дифференцируема в $x^0$ и $\nabla f(x^0) \neq 0$.
  Тогда среди всех единичных векторов $l$ производная по направлению
  максимальна в направлении градиента и минимальна в противоположном:
  \[
    \max_{\|l\|=1} \frac{\partial f}{\partial l}(x^0)
    = \|\nabla f(x^0)\|,
    \quad
    \text{достигается при }
    l = \frac{\nabla f(x^0)}{\|\nabla f(x^0)\|},
  \]
  \[
    \min_{\|l\|=1} \frac{\partial f}{\partial l}(x^0)
    = -\|\nabla f(x^0)\|,
    \quad
    \text{достигается при }
    l = -\frac{\nabla f(x^0)}{\|\nabla f(x^0)\|}.
  \]
  \begin{proof}
    По второму НУ для единичного $l$:
    \[
      \frac{\partial f}{\partial l}(x^0)
      = \langle \nabla f(x^0), l \rangle
      = \|\nabla f(x^0)\| \cdot \cos\theta,
    \]
    где $\theta$ --- угол между $\nabla f(x^0)$ и $l$.
    Так как $-1 \leq \cos\theta \leq 1$:
    \[
      -\|\nabla f(x^0)\|
      \;\leq\;
      \langle \nabla f(x^0), l \rangle
      \;\leq\;
      \|\nabla f(x^0)\|.
    \]
    При $l = \tfrac{\nabla f(x^0)}{\|\nabla f(x^0)\|}$
    получаем $\cos\theta = 1$ --- максимум достигнут.
    При $l = -\tfrac{\nabla f(x^0)}{\|\nabla f(x^0)\|}$
    получаем $\cos\theta = -1$ --- минимум достигнут. \qedhere
  \end{proof}
\end{theorem}
