\question{Предел точек в $\mathbb{R}^n$. Покоординатная сходимость. Теорема Больцано--Вейерштрасса}

\begin{defn}[Предел по совокупности по Коши]
  Пусть $E \subset \mathbb{R}^n$, $x^0$ --- предельная точка $E$, $f: E \to \mathbb{R}$.
  Будем говорить, что $C \in \mathbb{R}$ --- \emph{предел $f$ в точке $x^0$} и писать
  $\lim_{x \to x^0} f(x) = C$, если
  \[
    \forall\varepsilon>0\;\exists\delta>0:\;
    \forall x \in E,\; 0 < \|x-x^0\| < \delta \;\Rightarrow\; |f(x)-C|<\varepsilon.
  \]
\end{defn}

\begin{defn}[Предел по Гейне]
  $\lim_{x\to x^0} f(x) = C$, если для любой последовательности
  $x^{(m)} \in E$, $x^{(m)} \neq x^0$, $x^{(m)} \to x^0$:
  $\lim_{m\to\infty} f(x^{(m)}) = C$.
\end{defn}

\begin{theorem}[Эквивалентность Коши и Гейне]
  Оба определения равносильны.
  \begin{proof}
    \textbf{Коши $\Rightarrow$ Гейне.} Дано $\varepsilon>0$, берём $\delta$ из Коши.
    Так как $x^{(m)}\to x^0$, начиная с некоторого $M$: $\|x^{(m)}-x^0\|<\delta$, откуда $|f(x^{(m)})-C|<\varepsilon$.

    \textbf{Гейне $\Rightarrow$ Коши.} От противного: $\exists\varepsilon_0>0$: $\forall\delta>0\;\exists x_\delta$
    с $0<\|x_\delta-x^0\|<\delta$ и $|f(x_\delta)-C|\geq\varepsilon_0$.
    Берём $\delta=1/m$ --- получаем последовательность, противоречащую определению по Гейне. \qedhere
  \end{proof}
\end{theorem}

\begin{defn}[Предел по направлению]
  Для $l \in \mathbb{R}^n \setminus \{0\}$ пределом $f$ по вектору $l$ в точке $x^0$ называется
  $\lim_{t\to+0} f(x^0+tl)$, если он существует.
\end{defn}

\begin{note}
  Существование предела по совокупности влечёт существование предела по любому направлению
  (и все они равны). Обратное неверно.
  Контрпример: $f(x,y) = \dfrac{x^2 y}{x^4+y^2}$ в $(0,0)$.
  По любому линейному направлению $l=(l_1,l_2)$:
  $f(tl_1,tl_2) = \dfrac{t^4 l_1^2 l_2}{t^4 l_1^4 + t^4 l_2^2} \cdot \dfrac{1}{t^0} \to 0$
  (при $l_2\neq0$: $\to l_1^2 l_2 / l_2^2 \cdot 1/t^0$... фактически $\to 0$).
  Но вдоль кривой $y = x^2$: $f(x,x^2) = \dfrac{x^4}{2x^4} = \dfrac{1}{2}$. Предела нет.
\end{note}

\begin{theorem}[Связь покоординатной сходимости и сходимости по норме]
  \label{thm:coord_convergence}
  Последовательность $\{x^{(m)}\} \subset \mathbb{R}^n$ сходится к $x^0$ тогда и только тогда,
  когда каждая координата сходится:
  \[
    x^{(m)} \to x^0 \quad\Longleftrightarrow\quad
    x^{(m)}_k \to x^0_k \quad\forall k \in \{1,\ldots,n\}.
  \]
  \begin{proof}
    \textbf{$\Rightarrow$.} $|x^{(m)}_k - x^0_k| \leq \|x^{(m)}-x^0\| \to 0$.

    \textbf{$\Leftarrow$.} $\|x^{(m)}-x^0\|_\infty = \max_k |x^{(m)}_k - x^0_k| \to 0$
    $\Rightarrow \|x^{(m)}-x^0\| \to 0$ по эквивалентности норм в $\mathbb{R}^n$. \qedhere
  \end{proof}
\end{theorem}

\begin{note}
  \textbf{Предупреждение.} Теорема выше --- о сходимости \emph{последовательностей точек},
  а не о пределах функций. Существование всех итерированных пределов функции $f$ по каждой переменной
  \emph{не влечёт} существования предела по совокупности.
\end{note}

\begin{theorem}[Больцано--Вейерштрасс]
  Из любой ограниченной последовательности в $\mathbb{R}^n$ можно выделить сходящуюся подпоследовательность.
  \begin{proof}
    Индукция по $n$. При $n=1$ --- классическая теорема.
    Предположим утверждение для $n-1$. Проекция $\{x^{(m)}\}$ на первую координату ограничена в $\mathbb{R}$,
    выделяем подпоследовательность $\{x^{(m_j)}\}$, по которой первая координата сходится.
    Ограниченная последовательность в $\mathbb{R}^{n-1}$ (по оставшимся координатам) снова имеет
    сходящуюся подпоследовательность. Повторяем $n$ раз --- результат сходится покоординатно, откуда и по норме. \qedhere
  \end{proof}
\end{theorem}

\question{Бесконечная дифференцируемость суммы степенного ряда. Единственность разложения}

\begin{theorem}[Бесконечная дифференцируемость суммы степенного ряда]
  Пусть ряд $S(x) = \sum_{n=0}^\infty c_n(x-x_0)^n$ имеет радиус сходимости $R > 0$.
  Тогда $S \in C^\infty((x_0-R,\, x_0+R))$ и его можно дифференцировать почленно:
  \[
    S^{(k)}(x) = \sum_{n=k}^\infty c_n \frac{n!}{(n-k)!}(x-x_0)^{n-k}.
  \]
  \begin{proof}
    Для любого $[a,b] \subset (x_0-R, x_0+R)$ производный ряд равномерно сходится
    (мажорируется по признаку Вейерштрасса). По теореме о почленном дифференцировании равномерно
    сходящегося ряда из дифференцируемых функций, $S$ дифференцируема и $S' = (\text{производный ряд})$.
    Применяем индуктивно. \qedhere
  \end{proof}
\end{theorem}

\begin{theorem}[Единственность разложения в степенной ряд]
  Если $f(x) = \sum_{n=0}^\infty c_n(x-x_0)^n$ в окрестности $x_0$, то коэффициенты
  определены однозначно:
  \[
    c_n = \frac{f^{(n)}(x_0)}{n!}.
  \]
  \begin{proof}
    По предыдущей теореме $f^{(k)}(x_0) = k!\, c_k$. \qedhere
  \end{proof}
\end{theorem}
